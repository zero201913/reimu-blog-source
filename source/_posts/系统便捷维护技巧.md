# 一键安装常用软件
```bat
@echo off
chcp 65001 >nul
echo 使用Winget一键安装软件

:: 检查Winget是否可用
where winget >nul 2>&1
if %errorLevel% neq 0 (
    echo 错误：系统不支持Winget，请使用其他安装脚本
    pause
    exit /b 1
)

echo 开始安装常用软件...

:: 安装软件列表
winget install --id Google.Chrome -e --accept-package-agreements --accept-source-agreements
winget install --id Microsoft.Edge -e --accept-package-agreements --accept-source-agreements
winget install --id Tencent.WeChat -e --accept-package-agreements --accept-source-agreements
winget install --id Tencent.QQ -e --accept-package-agreements --accept-source-agreements
winget install --id 7zip.7zip -e --accept-package-agreements --accept-source-agreements
winget install --id Notepad++.Notepad++ -e --accept-package-agreements --accept-source-agreements
winget install --id Foxit.FoxitReader -e --accept-package-agreements --accept-source-agreements

echo.
echo ================================
echo       软件安装完成！
echo ================================
pause
```

## 通过useCookie来便捷设置


安装vue+vite命令
```cmd
npm init vite@latest test --template vue
```

```html
<template>
  <div>
    关于页面
    <el-button @click="set">设置</el-button>
    <el-button @click="get">读取</el-button>
    <el-button @click="del">删除</el-button>
  </div>
</template>

<script setup>
import { useCookies } from '@vueuse/integrations/useCookies';
const cookie = useCookies()
console.log('当前 cookie:', cookie);
function set() {
  cookie.set('test_cookie', 'hello world', { expires: new Date(Date.now() + 86400e3) });
  console.log('设置 cookie:', cookie.get('test_cookie'));
}
function get(){
  console.log('读取 cookie:', cookie.get('test_cookie'));
}

function del(){
  cookie.remove('test_cookie');
  console.log('删除 cookie:', cookie.get('test_cookie'));
}
</script>
```

## vue的数据同步

```vue
<template>
  <div>
    关于页面 
    <el-button @click="increment" type="primary" >{{ count }}</el-button> <!-- 模板自动解包，显示数值 -->
    <el-button @click="add" type="primary" >{{ form.age }}</el-button> <!-- 模板自动解包，显示数值 -->
    <!-- <Helloworld msg="Welcome to Your Vue.js + Vite App"/> -->
  </div>
</template>

<script setup>
import { ref,reactive } from 'vue' // 必须导入 ref
// import Helloworld from '../components/HelloWorld.vue'


let count = ref(1) 

function increment() {
  count.value++
}

const form = reactive({
  age: 0
})

function add(){
    form.age++
}

</script>
```
